Service Worker

Kelola caching dan offline mode

Pemilik

Service Worker (sw.js)

Memeriksa status service worker...

Service worker di bawah mengimplementasikan strategi cache-first untuk aset statis dan menyediakan halaman offline saat koneksi terputus. Letakkan kode ini di file sw.js di direktori utama aplikasi.

const CACHE_NAME = 'toko-grosir-v1.0.0'; const PRECACHE_ASSETS = [ '/', 'blok1-login.html', 'blok2-dasbor.html', 'blok3-profil.html', 'blok6-dasborkasir.html', 'css/style.css', 'js/app.js' ]; // Install event: pre-cache aset penting self.addEventListener('install', function(event) { event.waitUntil( caches.open(CACHE_NAME).then(function(cache) { console.log('Service Worker: Pre-caching aset'); return cache.addAll(PRECACHE_ASSETS); }) ); }); // Activate event: hapus cache lama self.addEventListener('activate', function(event) { event.waitUntil( caches.keys().then(function(cacheNames) { return Promise.all( cacheNames.filter(function(name) { return name !== CACHE_NAME; }).map(function(name) { console.log('Service Worker: Menghapus cache lama', name); return caches.delete(name); }) ); }) ); }); // Fetch event: strategi cache-first, fallback ke network self.addEventListener('fetch', function(event) { event.respondWith( caches.match(event.request).then(function(cachedResponse) { // Kembalikan dari cache jika ada if (cachedResponse) { return cachedResponse; } // Jika tidak ada di cache, ambil dari jaringan return fetch(event.request).then(function(response) { // Hanya cache response yang sukses if (!response || response.status !== 200 || response.type !== 'basic') { return response; } var responseToCache = response.clone(); caches.open(CACHE_NAME).then(function(cache) { cache.put(event.request, responseToCache); }); return response; }).catch(function() { // Jika offline dan bukan HTML, bisa tampilkan fallback if (event.request.headers.get('accept').includes('text/html')) { return caches.match('blok1-login.html'); } }); }) ); });
